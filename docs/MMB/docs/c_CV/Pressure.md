\(
	\text{推定エネルギー必要量 (EER)} \\
	= \text{基礎代謝量 (BMR)} \times \text{身体活動レベル} \\
	= \text{基礎代謝基準値} \times \text{基準体重} \times \text{身体活動レベル} \\
	= 21.5 _{\text{(kcal/kg/day)}} \times 22_{\text{(kg/m/m)}} \times \text{BH}^2 
	\begin{cases} 
		1.50 &\text{(I)} \\ 
		1.75 &\text{(II)} \\ 
		2.00 &\text{(III)} 
	\end{cases} \\
	\approx 22 \times \text{BH}^2 \times 
	\begin{cases}
		30 & \text{(I)}\\
		37.5 & \text{(II)} \\
		40 & \text{(III)}
	\end{cases}
\)
